132 #define HAL_THREAD_INIT_CONTEXT( __sparg, __thread, __entry, __id ) \
133 { \
134  register CYG_WORD __sp = ((CYG_WORD)__sparg) & ~7; \
135  register CYG_WORD *__ep = (CYG_WORD *)(__sp -= sizeof(CYG_WORD)); \
136  register HAL_SavedRegisters *__regs; \
137  int __i; \
138  __sp = ((CYG_WORD)__sp) &~15; \
139  __regs = (HAL_SavedRegisters *)((__sp) - sizeof(__regs->u.thread)); \
140  __regs->u.type = HAL_SAVEDREGISTERS_THREAD; \
141  for( __i = 1; __i < 13; __i++ ) \
142  __regs->u.thread.r[__i] = 0; \
143  *__ep = (CYG_WORD)(__entry); \
144  __regs->u.thread.sp = (CYG_WORD)(__sp); \
145  __regs->u.thread.r[0] = (CYG_WORD)(__thread); \
146  __regs->u.thread.r[1] = (CYG_WORD)(__id); \
147  __regs->u.thread.r[11] = (CYG_WORD)(__ep); \
148  __regs->u.thread.pc = (CYG_WORD)__entry; \
149  __regs->u.thread.basepri = 0; \
150  __sparg = (CYG_ADDRESS)__regs; \
151 }

300 // Minimal and sensible stack sizes: the intention is that applications
301 // will use these to provide a stack size in the first instance prior to
302 // proper analysis. Idle thread stack should be this big.
303 
304 // THESE ARE NOT INTENDED TO BE MICROMETRICALLY ACCURATE FIGURES.
305 // THEY ARE HOWEVER ENOUGH TO START PROGRAMMING.
306 // YOU MUST MAKE YOUR STACKS LARGER IF YOU HAVE LARGE "AUTO" VARIABLES!
307 
308 // This is not a config option because it should not be adjusted except
309 // under "enough rope" sort of disclaimers.
310 
311 // A minimal, optimized stack frame - space for return link plus four
312 // arguments or local variables.
313 #define CYGNUM_HAL_STACK_FRAME_SIZE (4 * 20)
314 
315 // Stack needed for a context switch
316 #define CYGNUM_HAL_STACK_CONTEXT_SIZE (4 * 20)
317 
318 // Interrupt + call to ISR, interrupt_end() and the DSR
319 #define CYGNUM_HAL_STACK_INTERRUPT_SIZE \
320  (CYGNUM_HAL_STACK_CONTEXT_SIZE + 2 * CYGNUM_HAL_STACK_FRAME_SIZE)
321 
322 // Space for the maximum number of nested interrupts, plus room to call functions
323 #define CYGNUM_HAL_MAX_INTERRUPT_NESTING 4
324 
325 // Minimum stack size. Space for the given number of nested
326 // interrupts, plus a thread context switch plus a couple of function
327 // calls.
328 #define CYGNUM_HAL_STACK_SIZE_MINIMUM \
329  ((CYGNUM_HAL_MAX_INTERRUPT_NESTING+1) * CYGNUM_HAL_STACK_INTERRUPT_SIZE + \
330  2 * CYGNUM_HAL_STACK_FRAME_SIZE)
331 
332 // Typical stack size -- used mainly for test programs. The minimum
333 // stack size plus enough space for some function calls.
334 #define CYGNUM_HAL_STACK_SIZE_TYPICAL \
335  (CYGNUM_HAL_STACK_SIZE_MINIMUM + 32 * CYGNUM_HAL_STACK_FRAME_SIZE)
