107 Cyg_Thread *
108 Cyg_Scheduler_Implementation::schedule(void)
109 {
110  CYG_REPORT_FUNCTYPE("returning thread %08x");
111 
112  // The run queue may _never_ be empty, there is always
113  // an idle thread at the lowest priority.
114 
115  CYG_ASSERT( queue_map != 0, "Run queue empty");
116  CYG_ASSERT( queue_map & (1<<CYG_THREAD_MIN_PRIORITY), "Idle thread vanished!!!");
117  CYG_ASSERT( !run_queue[CYG_THREAD_MIN_PRIORITY].empty(), "Idle thread vanished!!!");
118 
119 
120  register cyg_uint32 index;
121 
122  HAL_LSBIT_INDEX(index, queue_map);
123 
124  Cyg_RunQueue *queue = &run_queue[index];
125 
126  CYG_ASSERT( !queue->empty(), "Queue for index empty");
127 
128  Cyg_Thread *thread = queue->get_head();
129 
130 
131 
132  CYG_INSTRUMENT_MLQ( SCHEDULE, thread, index);
133 
134  CYG_ASSERT( thread != NULL , "No threads in run queue");
135  CYG_ASSERT( thread->queue == NULL , "Runnable thread on a queue!");
136 
137  CYG_REPORT_RETVAL(thread);
138 
139  return thread;
140 }

144 void
145 Cyg_Scheduler_Implementation::add_thread(Cyg_Thread *thread)
146 {
147  CYG_REPORT_FUNCTION();
148  CYG_REPORT_FUNCARG1("thread=%08x", thread);
149 
150  cyg_priority pri = thread->priority;
151  Cyg_RunQueue *queue = &run_queue[pri];
152 
153  CYG_INSTRUMENT_MLQ( ADD, thread, pri);
154 
155  CYG_ASSERT((CYG_THREAD_MIN_PRIORITY >= pri)
156  && (CYG_THREAD_MAX_PRIORITY <= pri),
157  "Priority out of range!");
158 
159  CYG_ASSERT( ((queue_map & (1<<pri))!=0) == ((!run_queue[pri].empty())!=0), "Map and queue disagree");
160 
161  // If the thread is on some other queue, remove it
162  // here.
163  if( thread->queue != NULL )
164  {
165  thread->queue->remove(thread);
166  }
167 
168  if( queue->empty() )
169  {
170  // set the map bit and ask for a reschedule if this is a
171  // new highest priority thread.
172 
173  queue_map |= (1<<pri);
174 
175  }
176  // else the queue already has an occupant, queue behind him
177 
178  queue->add_tail(thread);
179 
180  // If the new thread is higher priority than any
181  // current thread, request a reschedule.
182 
183  set_need_reschedule(thread);
184 
185  // Also reset the timeslice_count so that this thread gets a full
186  // timeslice once it begins to run.
187 
188  thread->timeslice_reset();
189 
190 
191  CYG_ASSERT( thread->queue == NULL , "Runnable thread on a queue!");
192  CYG_ASSERT( queue_map != 0, "Run queue empty");
193  CYG_ASSERT( queue_map & (1<<pri), "Queue map bit not set for pri");
194  CYG_ASSERT( !run_queue[pri].empty(), "Queue for pri empty");
195  CYG_ASSERT( ((queue_map & (1<<pri))!=0) == ((!run_queue[pri].empty())!=0), "Map and queue disagree");
196  CYG_ASSERT( queue_map & (1<<CYG_THREAD_MIN_PRIORITY), "Idle thread vanished!!!");
197  CYG_ASSERT( !run_queue[CYG_THREAD_MIN_PRIORITY].empty(), "Idle thread vanished!!!");
198 
199  CYG_REPORT_RETURN();
200 }

205 Cyg_Scheduler_Implementation::rem_thread(Cyg_Thread *thread)
206 {
207  CYG_REPORT_FUNCTION();
208  CYG_REPORT_FUNCARG1("thread=%08x", thread);
209 
210  CYG_ASSERT( queue_map != 0, "Run queue empty");
211 
212  cyg_priority pri = thread->priority;
213  Cyg_RunQueue *queue = &run_queue[pri];
214 
215  CYG_INSTRUMENT_MLQ( REM, thread, pri);
216 
217  CYG_ASSERT( pri != CYG_THREAD_MIN_PRIORITY, "Idle thread trying to sleep!");
218  CYG_ASSERT( !run_queue[CYG_THREAD_MIN_PRIORITY].empty(), "Idle thread vanished!!!");
219 
220 
221  CYG_ASSERT( queue_map & (1<<pri), "Queue map bit not set for pri");
222  CYG_ASSERT( !run_queue[pri].empty(), "Queue for pri empty");
223 
224  // remove thread from queue
225  queue->remove(thread);
226 
227  if( queue->empty() )
228  {
229  // If this was only thread in
230  // queue, clear map.
231 
232  queue_map &= ~(1<<pri);
233  }
234 
235  CYG_ASSERT( queue_map != 0, "Run queue empty");
236  CYG_ASSERT( queue_map & (1<<CYG_THREAD_MIN_PRIORITY), "Idle thread vanished!!!");
237  CYG_ASSERT( !run_queue[CYG_THREAD_MIN_PRIORITY].empty(), "Idle thread vanished!!!");
238  CYG_ASSERT( ((queue_map & (1<<pri))!=0) == ((!run_queue[pri].empty())!=0), "Map and queue disagree");
239 
240  CYG_REPORT_RETURN();
241 }

330 void
331 Cyg_Scheduler_Implementation::timeslice_cpu(void)
332 {
333 
334 
335  Cyg_Thread *thread = get_current_thread();
336  HAL_SMP_CPU_TYPE cpu_this = CYG_KERNEL_CPU_THIS();
337 
338  CYG_ASSERT( queue_map != 0, "Run queue empty");
339  CYG_ASSERT( queue_map & (1<<CYG_THREAD_MIN_PRIORITY), "Idle thread vanished!!!");
340 
341 
342  if( thread->timeslice_enabled &&
343  timeslice_count[cpu_this] == 0 )
344 
345  if( timeslice_count[cpu_this] == 0 )
346 
347  {
348  CYG_INSTRUMENT_SCHED(TIMESLICE,0,0);
349 
350 
351  CYG_ASSERT( get_sched_lock() > 0 , "Timeslice called with zero sched_lock");
352 
353  // Only try to rotate the run queue if the current thread is running.
354  // Otherwise we are going to reschedule anyway.
355  if( thread->get_state() == Cyg_Thread::RUNNING )
356  {
357  Cyg_Scheduler *sched = &Cyg_Scheduler::scheduler;
358 
359  CYG_INSTRUMENT_MLQ( TIMESLICE, thread, 0);
360 
361  CYG_ASSERTCLASS( thread, "Bad current thread");
362  CYG_ASSERTCLASS( sched, "Bad scheduler");
363 
364  cyg_priority pri = thread->priority;
365  Cyg_RunQueue *queue = &sched->run_queue[pri];
366 
367 
368  queue->rotate();
369 
370 
371  if( queue->get_head() != thread )
372  sched->set_need_reschedule();
373 
374  timeslice_count[cpu_this] = CYGNUM_KERNEL_SCHED_TIMESLICE_TICKS;
375  }
376  }
377 
378 
379  CYG_ASSERT( queue_map & (1<<CYG_THREAD_MIN_PRIORITY), "Idle thread vanished!!!");
380  CYG_ASSERT( !run_queue[CYG_THREAD_MIN_PRIORITY].empty(), "Idle thread vanished!!!");
381 #ifdef CYGDBG_KERNEL_TRACE_TIMESLICE
382  CYG_REPORT_RETURN();
383 #endif
384 }

419 
420 void
421 Cyg_SchedThread_Implementation::yield(void)
422 {
423  CYG_REPORT_FUNCTION();
424 
425  // Prevent preemption
426  Cyg_Scheduler::lock();
427 
428  Cyg_Thread *thread = CYG_CLASSFROMBASE(Cyg_Thread,
429  Cyg_SchedThread_Implementation,
430  this);
431 
432  // Only do this if this thread is running. If it is not, there
433  // is no point.
434 
435  if( thread->get_state() == Cyg_Thread::RUNNING )
436  {
437  // To yield we simply rotate the appropriate
438  // run queue to the next thread and reschedule.
439 
440  CYG_INSTRUMENT_MLQ( YIELD, thread, 0);
441 
442  CYG_ASSERTCLASS( thread, "Bad current thread");
443 
444  Cyg_Scheduler *sched = &Cyg_Scheduler::scheduler;
445 
446  CYG_ASSERTCLASS( sched, "Bad scheduler");
447 
448  cyg_priority pri = thread->priority;
449  Cyg_RunQueue *queue = &sched->run_queue[pri];
450 
451 
452  queue->rotate();
453 
454  if( queue->get_head() != thread )
455  sched->set_need_reschedule();
456  else
457  {
458  // Reset the timeslice counter so that this thread gets a
459  // full quantum as a reward for yielding when it is
460  // eventually rescheduled.
461  thread->timeslice_reset();
462  }
463 
464  }
465 
466  // Unlock the scheduler and switch threads
467 #ifdef CYGDBG_USE_ASSERTS
468  // This test keeps the assertions in unlock_inner() happy if
469  // need_reschedule was not set above.
470  if( !Cyg_Scheduler::get_need_reschedule() )
471  Cyg_Scheduler::unlock();
472  else
473 #endif
474  Cyg_Scheduler::unlock_reschedule();
475 
476 
477  CYG_REPORT_RETURN();
478 }

553 void
554 Cyg_ThreadQueue_Implementation::enqueue(Cyg_Thread *thread)
555 {
556  CYG_REPORT_FUNCTION();
557  CYG_REPORT_FUNCARG1("thread=%08x", thread);
558 
559  CYG_INSTRUMENT_MLQ( ENQUEUE, this, thread );
560 
561 #ifdef CYGIMP_KERNEL_SCHED_SORTED_QUEUES
562 
563  // Insert the thread into the queue in priority order.
564 
565  Cyg_Thread *qhead = get_head();
566 
567  if( qhead == NULL ) add_tail( thread );
568  else if( qhead == qhead->get_next() )
569  {
570  // There is currently only one thread in the queue, join it
571  // and adjust the queue pointer to point to the highest
572  // priority of the two. If they are the same priority,
573  // leave the pointer pointing to the oldest.
574 
575  qhead->insert( thread );
576 
577  if( thread->priority < qhead->priority )
578  to_head(thread);
579  }
580  else
581  {
582  // There is more than one thread in the queue. First check
583  // whether we are of higher priority than the head and if
584  // so just jump in at the front. Also check whether we are
585  // lower priority than the tail and jump onto the end.
586  // Otherwise we really have to search the queue to find
587  // our place.
588 
589  if( thread->priority < qhead->priority )
590  {
591  qhead->insert( thread );
592  to_head(thread);
593  }
594  else if( thread->priority > get_tail()->priority )
595  {
596  // We are lower priority than any thread in the queue,
597  // go in at the end.
598 
599  add_tail( thread );
600  }
601  else
602  {
603  // Search the queue. We do this backwards so that we
604  // always add new threads after any that have the same
605  // priority.
606 
607  // Because of the previous tests we know that this
608  // search will terminate before we hit the head of the
609  // queue, hence we do not need to check for that
610  // condition.
611 
612  Cyg_Thread *qtmp = get_tail();
613 
614  // Scan the queue until we find a higher or equal
615  // priority thread.
616 
617  while( qtmp->priority > thread->priority )
618  qtmp = qtmp->get_prev();
619 
620  // Append ourself after the node pointed to by qtmp.
621 
622  qtmp->append( thread );
623  }
624  }
625 #else
626  // Just add the thread to the tail of the list
627  add_tail( thread );
628 #endif
629 
630  thread->queue = CYG_CLASSFROMBASE(Cyg_ThreadQueue,
631  Cyg_ThreadQueue_Implementation,
632  this);
633  CYG_REPORT_RETURN();
634 }
