110  {
111  CYG_ASSERT( lock == 0, "spinlock still claimed");
112  };
113 
114  // Spin on the lock. In this case we just set it to 1 and proceed.
115  void spin()
116  {
117  CYG_ASSERT( lock == 0, "spinlock already claimed!");
118  lock = 1;
119  };
120 
121  // Clear the lock. Again, just set the value.
122  void clear()
123  {
124  CYG_ASSERT( lock != 0, "spinlock already cleared!");
125  lock = 0;
126  };
127 
128  // Try to claim the lock. Return true if successful, false if not.
129  cyg_bool trylock()
130  {
131  if( lock ) return false;
132  else { lock = 1; return true; }
133  };
134 
135  // Test the current value of the lock
136  cyg_bool test() { return lock; };
137 
138 
139  // The following two member functions are only necessary if the
140  // spinlock is to be used in an ISR.
141 
142  // Claim the spinlock, but also mask this CPU's interrupts while
143  // we have it.
144  void spin_intsave(CYG_INTERRUPT_STATE *state)
145  {
146  CYG_INTERRUPT_STATE s;
147  HAL_DISABLE_INTERRUPTS(s);
148  *state = s;
149  spin();
150  };
151 
152  // Clear the lock, and restore the interrupt state saved in
153  // spin_intsave().
154  void clear_intsave(CYG_INTERRUPT_STATE state)
