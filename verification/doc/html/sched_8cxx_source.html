106 // This is only called when there is the potential for real work to be
107 // done. Other cases are handled in Cyg_Scheduler::unlock() which is
108 // an inline; _or_ this function may have been called from
109 // Cyg_Scheduler::reschedule(), or Cyg_Scheduler::unlock_reschedule. The
110 // new_lock argument contains the value that the scheduler lock should
111 // have after this function has completed. If it is zero then the lock is
112 // being released and some extra work (running ASRs, checking for DSRs) is
113 // done before returning. If it is non-zero then it must equal the
114 // current value of the lock, and is used to indicate that we want to
115 // reacquire the scheduler lock before returning. This latter option
116 // only makes any sense if the current thread is no longer runnable,
117 // e.g. sleeping, otherwise this function will do nothing.
118 // This approach of passing in the lock value at the end effectively
119 // makes the scheduler lock a form of per-thread variable. Each call
120 // to unlock_inner() carries with it the value the scheduler should
121 // have when it reschedules this thread back, and leaves this function.
122 // When it is non-zero, and the thread is rescheduled, no ASRS are run,
123 // or DSRs processed. By doing this, it makes it possible for threads
124 // that want to go to sleep to wake up with the scheduler lock in the
125 // same state it was in before.

306 void Cyg_Scheduler::thread_entry( Cyg_Thread *thread )
307 {
308  clear_need_reschedule(); // finished rescheduling
309  set_current_thread(thread); // restore current thread pointer
310 
311  CYG_INSTRUMENT_THREAD(ENTER,thread,0);
312 
313  thread->timeslice_reset();
314  thread->timeslice_restore();
315 
316  // Finally unlock the scheduler. As well as clearing the scheduler
317  // lock this allows any pending DSRs to execute. The new thread
318  // must start with a lock of zero, so we keep unlocking until the
319  // lock reaches zero.
320  while( get_sched_lock() != 0 )
321  unlock();
322 }

507

509  // protocols. The simplification in this algorithm is that we do
510  // not reduce our priority until we have freed all mutexes
511  // claimed. Hence we can continue to run at an artificially high
512  // priority even when we should not. However, since nested
513  // mutexes are rare, the thread we have inherited from is likely
514  // to be locking the same mutexes we are, and mutex claim periods

516  // and a more complex algorithm should be negligible. The most
517  // important advantage of this algorithm is that it is fast and
518  // deterministic.

521  Cyg_SchedThread,

525

527  {
528  priority_inherited = false;

531  if( priority < original_priority )
532  self->set_priority( original_priority );
