236 /* --------------------------------------------------------------------------
237  * ba_context_process
238  *
239  * Bus administrator interrupts context process
240  *
241  * @param : UNSIGNED8 sf_f_code
242  * UNSIGNED16 sf_content_16
243  * UNSIGNED8 vector
244  * @return : void
245  * @condition : current master
246  * @see : devices_scan, mastership_transfer, md_context_process
247  *
248  * When current master receive multiple different interrupts, this function
249  * help to distribute interrupt vector and frame data to handle function
250  * according to bus administrator context.
251  *
252  * ba_context (current ba context):
253  * 01. BA_CONTEXT_NOTRUNNING : not current master, exit;
254  * 02. BA_CONTEXT_DEVICES_SCAN : devices scan;
255  * 03. BA_CONTEXT_MASTERSHIP_TRANSFER : mastership transfer;
256  * 04. BA_CONTEXT_MESSAGE_ARBITRATION : message arbitration.
257  * --------------------------------------------------------------------------
258  */
259 void ba_context_process(UNSIGNED8 sf_f_code, UNSIGNED16 sf_content_16, UNSIGNED8 vector)
260 {
261 #ifdef MVBPROJ_DEBUG_INFO
262  printf("[BA] Context process called.\n");
263 #endif
264  if (ba_context == BA_CONTEXT_NOTRUNNING)
265  return;
266 
267  UNSIGNED8 rs_code;
268  switch (ba_context)
269  {
270  case BA_CONTEXT_DEVICES_SCAN:
271  rs_code = RS_CODE_DEVICES_SCAN_FIN;
272 #ifndef MVBPROJ_DEVICES_SCAN_BLOCKED
273  rs_code = devices_scan(sf_f_code, sf_content_16, vector);
274 #endif // ifndef MVBPROJ_DEVICES_SCAN_BLOCKED
275  if (rs_code == RS_CODE_DEVICES_SCAN_FIN)
276  {
277  ba_context = BA_CONTEXT_MESSAGE_ARBITRATION;
278 #ifndef MVBPROJ_ARBITRATION_BLOCKED
279  md_context_process(EXTI_SUPERVISORY_INT_VECTOR);
280 #endif // ifndef MVBPROJ_MASTERSHIP_TRANSFER_BLOCKED
281  }
282  break;
283  case BA_CONTEXT_MASTERSHIP_TRANSFER:
284 #ifndef MVBPROJ_MASTERSHIP_TRANSFER_BLOCKED
285 #ifdef MVBPROJ_DEBUG_CONN_BA
286 cyg_interrupt_disable();
287 printf("bp_number: %d\n", bp_number);
288 cyg_interrupt_enable();
289 #endif
290  rs_code = mastership_transfer(sf_f_code, sf_content_16, vector);
291 #endif // ifndef MVBPROJ_MASTERSHIP_TRANSFER_BLOCKED
292  break;
293  case BA_CONTEXT_MESSAGE_ARBITRATION:
294 #ifndef MVBPROJ_ARBITRATION_BLOCKED
295  md_context_process(vector);
296 #endif // ifndef MVBPROJ_MASTERSHIP_TRANSFER_BLOCKED

330 /* --------------------------------------------------------------------------
331  * mastership_transfer
332  *
333  * Mastrship transfer function
334  *
335  * @param : UNSIGNED8 sf_f_code
336  * UNSIGNED16 sf_content_16
337  * UNSIGNED8 vector
338  * @return : UNSIGNED8 rs_code
339  * @condition : current master
340  *
341  * At the end of macro cycle, current master shall try to pass mastership to
342  * other backup administrator, or keep mastership when no other
343  * administrator available. This function should be called only once at each
344  * macro cycle.
345  *
346  * ba_context_mastership_transfer_act (current action context):
347  * 01. BA_CONTEXT_MASTERSHIP_TRANSFER_NULL :
348  * First action, ready to send device scan frame to scan device
349  * address ba_context_mastership_transfer_address.
350  * 02. BA_CONTEXT_MASTERSHIP_TRANSFER_DEVICE_STATUS :
351  * Wait for device status slave frame. If status.ba == 1 &&
352  * status.cs2 == 1, offer mastership and change context to
353  * BA_CONTEXT_MASTERSHIP_TRANSFER_OFFER_MASTERSHIP. If not, return to
354  * context BA_CONTEXT_MASTERSHIP_TRANSFER_NULL to send another device
355  * scan main frame.
356  * 03. BA_CONTEXT_MASTERSHIP_TRANSFER_OFFER_MASTERSHIP
357  * Wait for backup administrator to response F_Code = 8 main frame.
358  * Slave frame with the most significant bit as 1 means transfer
359  * succeed. Others will lead to keep mastership for current master.
360  * --------------------------------------------------------------------------
361  */
362 UNSIGNED8 mastership_transfer(UNSIGNED8 sf_f_code, UNSIGNED16 sf_content_16, UNSIGNED8 vector)
363 {
364 
365  if (vector == EXTI_SUPERVISORY_INT_VECTOR)

372  while (ba_context_mastership_transfer_address != this_addr)

375  {
376  break;
377  }

383 #ifdef MVBPROJ_DEBUG_CONN_BA

392  }
393

519 /* --------------------------------------------------------------------------
520  * devices_scan
521  *
522  * Devices scan function
523  *
524  * @param : UNSIGNED8 sf_f_code
525  * UNSIGNED16 sf_content_16
526  * UNSIGNED8 vector
527  * @return : UNSIGNED8 rs_code
528  * @condition : current master
529  * @see : build_list_of_devices_scan
530  *
531  * Each basic period (except last one in macro cycle), this function send
532  * several device scan frame (F_code = 15) according to the config in binary
533  * file (may be 0 or 1). In the administrator data structure
534  * MVB_ADMINISTRATOR, there is two strategy of devices scan, for all address
535  * and known devices (Also, there's known devices list in MVB_ADMINISTRATOR).
536  * To supervisory the bus, master should poll the status of each device, and
537  * build devices_list to store device status. Device not response for three
538  * times will be deleted from devices_list.
539  * --------------------------------------------------------------------------
540  */
541 UNSIGNED8 devices_scan(UNSIGNED8 sf_f_code, UNSIGNED16 sf_content_16, UNSIGNED8 vector)
542 {
543  if (device_scan_unit == 0)
544  return RS_CODE_DEVICES_SCAN_FIN;
545 
546  if (vector == EXTI_SUPERVISORY_INT_VECTOR)
547  {
548  ba_context_devices_scan_j = 0;
549  }
550 
551 
552  if (vector == EXTI_TIMEOUT_VECTOR)
553  {
554  build_list_of_devices_scan(FRAME_ADDRESS(current_mf), 0, true);
555  ba_context_devices_scan_j ++;
556 
557  }
558  else if (vector == EXTI_SLAVE_FRAME_VECTOR)
559  {
560  build_list_of_devices_scan(FRAME_ADDRESS(current_mf), sf_content_16, false);
561  ba_context_devices_scan_j ++;
562  }
563 
564  if (ba_context_devices_scan_j >= device_scan_unit)
565  return RS_CODE_DEVICES_SCAN_FIN;
566 
567  UNSIGNED32 KDL_BASE, KDL_CURRENT_ENTRY;
568  UNSIGNED32 count, *entry;
569  switch (devices_scan_strategy)
570  {
571  case DEVICES_SCAN_STRATEGY_ENUM_KNOWN:
572  KDL_BASE = (UNSIGNED32) MVB_ADMINISTRATOR_GET_P_KDL(admin_obj);
573  count = MVB_ADMINISTRATOR_GET_C_KDL(admin_obj);
574 
575  if (ba_context_devices_scan_i >= count)
576  ba_context_devices_scan_i = 0;
577 
578  KDL_CURRENT_ENTRY = KDL_BASE + ba_context_devices_scan_i * MVB_SIZEOF_STRUCT_MVB_ADMINISTRATOR_KDL;
579  entry = (UNSIGNED32 *)KDL_CURRENT_ENTRY;
580 
581  ba_context_devices_scan_address = *entry;
582 
583 #ifdef MVBPROJ_DEBUG_INFO
584  printf("\t[DEVICES SCAN] F=15 Address=0x%04x (KNOWN)\n", ba_context_devices_scan_address);
585 #endif
586  mvb_arm_send_main_frame_f(FRAME_F_CODE_15, ba_context_devices_scan_address);
587  ba_context_devices_scan_i ++;
588  break;
589  case DEVICES_SCAN_STRATEGY_ENUM_ALL:
590  if (ba_context_devices_scan_address >= 0x1000 || ba_context_devices_scan_address <= 0x0000)
591  ba_context_devices_scan_address = 0x0001;
592 
593 #ifdef MVBPROJ_DEBUG_INFO
594  printf("\t[DEVICES SCAN] F=15 Address=0x%04x (ALL)\n", ba_context_devices_scan_address);
